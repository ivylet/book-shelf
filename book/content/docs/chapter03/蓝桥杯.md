```c++
vector<int> preSum(vector<int> arr,int k){
	vector<int> pre(k+1);
	for(int i = 1; i < k+1;i++){
		pre[i] = pre[i - 1] + arr[i];
	}
}
```

```cpp
cha[i] = arr[i] - arr[i-1]
```
